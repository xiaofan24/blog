AnyOf ​

GitHub | Playground

如果数组为空，返回 false

type Sample2 = AnyOf<[0, '', false, [], {}]> // expected to be false.

InorderTraversal ​

GitHub | Playground

ts
const tree1 = {
  val: 1,
  left: null,
  right: {
    val: 2,
    left: {
      val: 3,
      left: null,
      right: null,
    },
    right: null,
  },
} as const

type A = InorderTraversal<typeof tree1> // [1, 3, 2]
